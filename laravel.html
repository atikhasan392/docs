php artisan key:generate

Model, Controller, Middleware, Mail, Event, Job, Work, Migration

php artisan make:controller UserController
php artisan make:middleware MinifyHTML
php artisan make:mail ContactMail
php artisan make:event UserStore
php artisan make:job SendEmailJob
php artisan queue:work

Databases & Migrations

php artisan migrate

php artisan cache:clear
php artisan config:clear
php artisan route:clear
php artisan view:clear
php artisan event:clear
php artisan optimize:clear

Others

Manual Storage Symlink

Route::get('/symlink', function() {
    $targetDir = storage_path('app/public');
    $linkDir = $_SERVER['DOCUMENT_ROOT'] . '/storage';
    symlink($targetDir, $linkDir);
});
